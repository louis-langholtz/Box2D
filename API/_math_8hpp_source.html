154 constexpr inline bool almost_equal(float x, float y, int ulp = 2)
155 {
156  // From http://en.cppreference.com/w/cpp/types/numeric_limits/epsilon :
157  // "the machine epsilon has to be scaled to the magnitude of the values used
158  // and multiplied by the desired precision in ULPs (units in the last place)
159  // unless the result is subnormal".
160  // Where "subnormal" means almost zero.
161  //
162  return (Abs(x - y) < (std::numeric_limits<float>::epsilon() * Abs(x + y) * ulp)) || almost_zero(x - y);
163 }

364 template <class T1, class T2>
365 constexpr inline auto Cross(const T1 a, const T2 b) noexcept
366 {
367  // Both vectors of same direction...
368  // If a = Vec2{1, 2} and b = Vec2{1, 2} then: a x b = 1 * 2 - 2 * 1 = 0.
369  // If a = Vec2{1, 2} and b = Vec2{2, 4} then: a x b = 1 * 4 - 2 * 2 = 0.
370  //
371  // Vectors at +/- 90 degrees of each other...
372  // If a = Vec2{1, 2} and b = Vec2{-2, 1} then: a x b = 1 * 1 - 2 * (-2) = 1 + 4 = 5.
373  // If a = Vec2{1, 2} and b = Vec2{2, -1} then: a x b = 1 * (-1) - 2 * 2 = -1 - 4 = -5.
374  //
375  // Vectors between 0 and 180 degrees of each other excluding 90 degrees...
376  // If a = Vec2{1, 2} and b = Vec2{-1, 2} then: a x b = 1 * 2 - 2 * (-1) = 2 + 2 = 4.
377  return (GetX(a) * GetY(b)) - (GetY(a) * GetX(b));
378 }

425 constexpr Vec2 Solve(const Mat22 mat, const Vec2 b) noexcept
426 {
427  const auto cp = Cross(mat.ex, mat.ey);
428  return (cp != 0)?
429  Vec2{(mat.ey.y * b.x - mat.ey.x * b.y) / cp, (mat.ex.x * b.y - mat.ex.y * b.x) / cp}:
430  Vec2{0, 0};

457 constexpr Vec3 Solve33(const Mat33& mat, const Vec3 b) noexcept
458 {
459  const auto dp = Dot(mat.ex, Cross(mat.ey, mat.ez));
460  const auto det = (dp != 0)? 1 / dp: dp;
461  const auto x = det * Dot(b, Cross(mat.ey, mat.ez));
462  const auto y = det * Dot(mat.ex, Cross(b, mat.ez));
463  const auto z = det * Dot(mat.ex, Cross(mat.ey, b));
464  return Vec3{x, y, z};
465 }

470 constexpr Vec2 Solve22(const Mat33& mat, const Vec2 b) noexcept
471 {
472  const auto cp = mat.ex.x * mat.ey.y - mat.ey.x * mat.ex.y;
473  const auto det = (cp != 0)? 1 / cp: cp;
474  const auto x = det * (mat.ey.y * b.x - mat.ey.x * b.y);
475  const auto y = det * (mat.ex.x * b.y - mat.ex.y * b.x);
476  return Vec2{x, y};
477 }

483 constexpr inline Mat33 GetInverse22(const Mat33& value) noexcept
484 {
485  const auto a = value.ex.x, b = value.ey.x, c = value.ex.y, d = value.ey.y;
486  auto det = (a * d) - (b * c);
487  if (det != RealNum{0})
488  {
489  det = RealNum{1} / det;
490  }
491  return Mat33{Vec3{det * d, -det * c, RealNum{0}}, Vec3{-det * b, det * a, 0}, Vec3{0, 0, 0}};
492 }

496 constexpr inline Mat33 GetSymInverse33(const Mat33& value) noexcept
497 {
498  auto det = Dot(value.ex, Cross(value.ey, value.ez));
499  if (det != RealNum{0})
500  {
501  det = RealNum{1} / det;
502  }
503 
504  const auto a11 = value.ex.x, a12 = value.ey.x, a13 = value.ez.x;
505  const auto a22 = value.ey.y, a23 = value.ez.y;
506  const auto a33 = value.ez.z;
507 
508  const auto ex_y = det * (a13 * a23 - a12 * a33);
509  const auto ey_z = det * (a13 * a12 - a11 * a23);
510  const auto ex_z = det * (a12 * a23 - a13 * a22);
511 
512  return Mat33{
513  Vec3{det * (a22 * a33 - a23 * a23), ex_y, ex_z},
514  Vec3{ex_y, det * (a11 * a33 - a13 * a13), ey_z},
515  Vec3{ex_z, ey_z, det * (a11 * a22 - a12 * a12)}
516  };
517 }
