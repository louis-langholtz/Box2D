487 template <typename T>
488 constexpr inline bool IsValid(const T& value) noexcept
489 {
490  // Note: This is not necessarily a no-op!! But it is a "constexpr".
491  //
492  // From http://en.cppreference.com/w/cpp/numeric/math/isnan:
493  // "Another way to test if a floating-point value is NaN is
494  // to compare it with itself:
495  // bool is_nan(double x) { return x != x; }
496  //
497  // So for all T, for which std::isnan() is implemented, this should work
498  // correctly and quite usefully!
499  //
500  return value == value;
501 }
