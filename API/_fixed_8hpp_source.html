91  template <typename T>
92  static constexpr value_type GetFromFloat(T val) noexcept
93  {
94  static_assert(std::is_floating_point<T>::value, "floating point value required");
95  // Note: std::isnan(val) *NOT* constant expression, so can't use here!
96  return !(val <= 0 || val >= 0)? GetNaN().m_value:
97  (val > static_cast<long double>(GetMax()))? GetInfinity().m_value:
98  (val < static_cast<long double>(GetLowest()))? GetNegativeInfinity().m_value:
99  static_cast<value_type>(val * ScaleFactor);
100  }

102  template <typename T>
103  static constexpr value_type GetFromSignedInt(T val) noexcept
104  {
105  static_assert(std::is_integral<T>::value, "integral value required");
106  static_assert(std::is_signed<T>::value, "must be signed");
107  return (val > (GetMax().m_value / ScaleFactor))? GetInfinity().m_value:
108  (val < (GetLowest().m_value / ScaleFactor))? GetNegativeInfinity().m_value:
109  val * ScaleFactor;
110  }

112  template <typename T>
113  static constexpr value_type GetFromUnsignedInt(T val) noexcept
114  {
115  static_assert(std::is_integral<T>::value, "integral value required");
116  static_assert(!std::is_signed<T>::value, "must be unsigned");
117  const auto max = static_cast<unsigned_wider_type>(GetMax().m_value / ScaleFactor);
118  return (val > max)? GetInfinity().m_value: static_cast<value_type>(val) * ScaleFactor;

292  constexpr Fixed& operator+= (Fixed val) noexcept
293  {
294  if (isnan() || val.isnan()
295  || ((m_value == GetInfinity().m_value) && (val.m_value == GetNegativeInfinity().m_value))
296  || ((m_value == GetNegativeInfinity().m_value) && (val.m_value == GetInfinity().m_value))
297  )
298  {
299  *this = GetNaN();
300  }
301  else if (val.m_value == GetInfinity().m_value)
302  {
303  m_value = GetInfinity().m_value;
304  }
305  else if (val.m_value == GetNegativeInfinity().m_value)
306  {
307  m_value = GetNegativeInfinity().m_value;
308  }
309  else if (isfinite() && val.isfinite())
310  {
311  const auto result = wider_type{m_value} + val.m_value;
312  if (result > GetMax().m_value)
313  {
314  // overflow from max
315  m_value = GetInfinity().m_value;
316  }
317  else if (result < GetLowest().m_value)
318  {
319  // overflow from lowest
320  m_value = GetNegativeInfinity().m_value;
321  }
322  else
323  {
324  m_value = static_cast<value_type>(result);
325  }
326  }
327  return *this;
328  }

368  constexpr Fixed& operator*= (Fixed val) noexcept
369  {
370  if (isnan() || val.isnan())
371  {
372  *this = GetNaN();
373  }
374  else if (!isfinite() || !val.isfinite())
375  {
376  if (m_value == 0 || val.m_value == 0)
377  {
378  *this = GetNaN();
379  }
380  else
381  {
382  *this = ((m_value > 0) != (val.m_value > 0))? -GetInfinity(): GetInfinity();
383  }
384  }
385  else
386  {
387  const auto product = wider_type{m_value} * wider_type{val.m_value};
388  const auto result = product / ScaleFactor;
389 
390  if (product != 0 && result == 0)
391  {
392  // underflow
393  m_value = static_cast<value_type>(result);
394  }
395  else if (result > GetMax().m_value)
396  {
397  // overflow from max
398  m_value = GetInfinity().m_value;
399  }
400  else if (result < GetLowest().m_value)
401  {
402  // overflow from lowest
403  m_value = GetNegativeInfinity().m_value;
404  }
405  else
406  {
407  m_value = static_cast<value_type>(result);
408  }
409  }
410  return *this;
411  }

413  constexpr Fixed& operator/= (Fixed val) noexcept
414  {
415  if (isnan() || val.isnan())

417  *this = GetNaN();
418  }
419  else if (!isfinite() && !val.isfinite())
420  {
421  *this = GetNaN();
422  }
423  else if (!isfinite())
