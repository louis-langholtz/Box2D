466  inline void Manifold::AddPoint(cf_t type, sidx_t index, Length2D point) noexcept
467  {
468  assert(m_pointCount < MaxManifoldPoints);
469  switch (m_type)
470  {
471  case e_unset:
472  break;
473  case e_circles:
474  break;
475  case e_faceA:
476  m_points[m_pointCount].localPoint = point;
477  m_points[m_pointCount].contactFeature.typeB = type;
478  m_points[m_pointCount].contactFeature.indexB = index;
479  ++m_pointCount;
480  break;
481  case e_faceB:
482  m_points[m_pointCount].localPoint = point;
483  m_points[m_pointCount].contactFeature.typeA = type;
484  m_points[m_pointCount].contactFeature.indexA = index;
485  ++m_pointCount;
486  break;
487  }
