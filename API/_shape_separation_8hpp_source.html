61  template <typename T1, typename T2>
62  static inline IndexSeparation GetMostAntiParallelSeparation(Span<const T1> points, const T2 refvec, const T1 offset)
63  {
64  // Search for the vector that is most anti-parallel to the reference vector.
65  // See: https://en.wikipedia.org/wiki/Antiparallel_(mathematics)#Antiparallel_vectors
66  auto index = IndexSeparation::InvalidIndex;
67  auto distance = IndexSeparation::InvalidDistance;
68  const auto count = points.size();
69  for (auto i = decltype(count){0}; i < count; ++i)
70  {
71  // Get cosine of angle between refvec and vectors[i] multiplied by their
72  // magnitudes (which will essentially be 1 for any two unit vectors).
73  // Get distance from offset to vectors[i] in direction of refvec.
74  const auto s = StripUnit(Dot(refvec, points[i] - offset));
75  if (distance > s)
76  {
77  distance = s;
78  index = static_cast<IndexSeparation::index_type>(i);
79  }
80  }
81  return IndexSeparation{distance, index};
82  }
