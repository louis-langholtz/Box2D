48  void Push(const T& element)
49  {
50  if (m_count == m_capacity)
51  {
52  T* old = m_stack;
53  m_capacity *= BufferGrowthRate;
54  m_stack = alloc<T>(m_capacity);
55  std::memcpy(m_stack, old, m_count * sizeof(T));
56  if (old != m_array)
57  {
58  free(old);
59  }
60  }
61 
62  m_stack[m_count] = element;
63  ++m_count;
64  }
65 
66  T Pop()
67  {
68  assert(m_count > 0);

71  }
72
