68  const_pointer find(Length2D value) const
69  {
70  // squaring anything smaller than the sqrt(std::numeric_limits<Vec2::data_type>::min())
71  // won't be reversible.
72  // i.e. won't obey the property that square(sqrt(a)) == a and sqrt(square(a)) == a.
73  for (auto&& elem: *this)
74  {
75  // length squared must be large enough to have a reasonable enough unit vector.
76  const auto delta = value - elem;
77  if (GetLengthSquared(StripUnits(delta)) <= m_minSepSquared)
78  {
79  // found or delta poorly conditioned
80  return &elem;
81  }
82  }
83  return end();
84  }
