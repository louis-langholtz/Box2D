672 {

1003 inline void SetForce(Body& body, const Force2D force, const Length2D point) noexcept
1004 {
1005  const auto linAccel = LinearAcceleration2D{force * body.GetInvMass()};
1006  const auto invRotI = body.GetInvRotInertia();
1007  const auto dp = point - body.GetWorldCenter();
1008  const auto cp = Torque{Cross(dp, force) / Radian};
1009  const auto angAccel = AngularAcceleration{cp * invRotI};
1010  body.SetAcceleration(linAccel, angAccel);
1011 }

1020 inline void ApplyForce(Body& body, const Force2D force, const Length2D point) noexcept
1021 {
1022  // Torque is L^2 M T^-2 QP^-1.
1023  const auto linAccel = LinearAcceleration2D{force * body.GetInvMass()};
1024  const auto invRotI = body.GetInvRotInertia(); // L^-2 M^-1 QP^2
1025  const auto dp = Length2D{point - body.GetWorldCenter()}; // L
1026  const auto cp = Torque{Cross(dp, force) / Radian}; // L * M L T^-2 is L^2 M T^-2
1027  const auto angAccel = AngularAcceleration{cp * invRotI}; // L^2 M T^-2 QP^-1 * L^-2 M^-1 QP^2 = QP T^-2;
1028  body.SetAcceleration(body.GetLinearAcceleration() + linAccel, body.GetAngularAcceleration() + angAccel);
1029 }

1042 inline void SetTorque(Body& body, const Torque torque) noexcept
1043 {
1044  const auto linAccel = body.GetLinearAcceleration();
1045  const auto invRotI = body.GetInvRotInertia();
1046  const auto intRotInertiaUnitless = invRotI * (SquareMeter * Kilogram / SquareRadian);
1047  const auto angAccel = RealNum{torque / NewtonMeter} * intRotInertiaUnitless * RadianPerSquareSecond;
1048  body.SetAcceleration(linAccel, angAccel);
1049 }
